//! Pluggable search backend trait for the web_search tool.

use anyhow::Result;

/// A single search result returned by a backend.
#[derive(Debug, Clone)]
pub struct SearchResult {
    pub title: String,
    pub url: String,
    pub snippet: String,
}

/// Trait for pluggable search backends.
/// Uses native async fn in traits (edition 2024).
pub trait SearchBackend: Send + Sync {
    /// Execute a search query and return up to `max_results` results.
    fn search(&self, query: &str, max_results: usize) -> impl std::future::Future<Output = Result<Vec<SearchResult>>> + Send;

    /// Backend name for logging and error messages.
    fn name(&self) -> &str;
}

/// Format search results into a string for the LLM.
pub fn format_results(results: &[SearchResult], query: &str) -> String {
    if results.is_empty() {
        return format!("No results found for: {query}");
    }
    let mut output = String::new();
    for (i, r) in results.iter().enumerate() {
        output.push_str(&format!(
            "{}. **{}**\n   {}\n   {}\n\n",
            i + 1,
            r.title,
            r.url,
            r.snippet,
        ));
    }
    output
}

//! SearXNG search backend — connects to a self-hosted SearXNG instance.

use anyhow::{Context, Result};
use super::search::{SearchBackend, SearchResult};

/// SearXNG backend configuration.
pub struct SearxngBackend {
    base_url: String,
    user_agent: String,
    client: reqwest::Client,
}

impl SearxngBackend {
    pub fn new(base_url: &str, user_agent: &str) -> Result<Self> {
        let client = reqwest::Client::builder()
            .timeout(std::time::Duration::from_secs(5))
            .user_agent(user_agent)
            .build()
            .context("Failed to build SearXNG HTTP client")?;
        Ok(Self {
            base_url: base_url.trim_end_matches('/').to_string(),
            user_agent: user_agent.to_string(),
            client,
        })
    }
}

impl SearchBackend for SearxngBackend {
    async fn search(&self, query: &str, max_results: usize) -> Result<Vec<SearchResult>> {
        let url = format!("{}/search", self.base_url);
        let response = self
            .client
            .get(&url)
            .query(&[
                ("q", query),
                ("format", "json"),
                ("language", "en"),
            ])
            .send()
            .await
            .context("SearXNG request failed")?;

        let status = response.status();
        if !status.is_success() {
            let text = response.text().await.unwrap_or_default();
            anyhow::bail!("SearXNG error ({status}): {text}");
        }

        let json: serde_json::Value = response
            .json()
            .await
            .context("Failed to parse SearXNG response")?;

        let results = json["results"]
            .as_array()
            .map(|arr| {
                arr.iter()
                    .take(max_results)
                    .filter_map(|r| {
                        Some(SearchResult {
                            title: r["title"].as_str()?.to_string(),
                            url: r["url"].as_str()?.to_string(),
                            snippet: r["content"].as_str().unwrap_or("").to_string(),
                        })
                    })
                    .collect()
            })
            .unwrap_or_default();

        Ok(results)
    }

    fn name(&self) -> &str {
        "searxng"
    }
}

/// Tavily hosted search backend.
pub struct TavilyBackend {
    base_url: String,
    api_key: String,
    client: reqwest::Client,
}

impl TavilyBackend {
    pub fn new(api_key: &str, base_url: Option<&str>) -> Self {
        let client = reqwest::Client::builder()
            .timeout(std::time::Duration::from_secs(30))
            .build()
            .unwrap_or_default();
        Self {
            base_url: base_url.unwrap_or("https://api.tavily.com").to_string(),
            api_key: api_key.to_string(),
            client,
        }
    }
}

impl SearchBackend for TavilyBackend {
    async fn search(&self, query: &str, max_results: usize) -> Result<Vec<SearchResult>> {
        let body = serde_json::json!({
            "query": query,
            "search_depth": "basic",
            "max_results": max_results,
            "include_answer": true,
        });

        let url = format!("{}/search", self.base_url);
        let response = self
            .client
            .post(&url)
            .bearer_auth(&self.api_key)
            .json(&body)
            .send()
            .await
            .context("Tavily request failed")?;

        let status = response.status();
        let text = response.text().await.unwrap_or_default();

        if !status.is_success() {
            anyhow::bail!("Tavily API error ({status}): {text}");
        }

        let json: serde_json::Value =
            serde_json::from_str(&text).context("Failed to parse Tavily response")?;

        let mut results = Vec::new();

        // Tavily provides an "answer" field — include it as the first result
        if let Some(answer) = json["answer"].as_str() {
            results.push(SearchResult {
                title: "AI Answer".to_string(),
                url: String::new(),
                snippet: answer.to_string(),
            });
        }

        if let Some(arr) = json["results"].as_array() {
            for r in arr.iter().take(max_results) {
                results.push(SearchResult {
                    title: r["title"].as_str().unwrap_or("Untitled").to_string(),
                    url: r["url"].as_str().unwrap_or("").to_string(),
                    snippet: r["content"].as_str().unwrap_or("").to_string(),
                });
            }
        }

        Ok(results)
    }

    fn name(&self) -> &str {
        "tavily"
    }
}

//! Web search tool — dispatches to pluggable search backends.

use anyhow::Result;
use serde_json::Value;

use super::search::{SearchBackend, format_results};
use super::search_searxng::SearxngBackend;
use super::search_tavily::TavilyBackend;
use crate::config::schema::ServiceConfig;
use crate::tools::ToolResult;

/// Build a search backend from service configuration.
pub fn build_backend(config: &ServiceConfig) -> Result<Box<dyn SearchBackend>> {
    match config.provider.as_str() {
        "searxng" => {
            let base_url = config
                .base_url
                .as_deref()
                .ok_or_else(|| anyhow::anyhow!("SearXNG requires 'base_url' in config"))?;
            let user_agent = config
                .user_agent
                .as_deref()
                .unwrap_or("Caboose/1.0 (SearchService)");
            Ok(Box::new(SearxngBackend::new(base_url, user_agent)?))
        }
        "tavily" => {
            let api_key = std::env::var(
                config
                    .api_key_env
                    .as_deref()
                    .unwrap_or("TAVILY_API_KEY"),
            )
            .map_err(|_| {
                anyhow::anyhow!(
                    "Tavily API key not found. Set the {} environment variable.",
                    config.api_key_env.as_deref().unwrap_or("TAVILY_API_KEY")
                )
            })?;
            Ok(Box::new(TavilyBackend::new(&api_key, config.base_url.as_deref())))
        }
        other => anyhow::bail!("Unsupported search provider: '{other}'. Supported: searxng, tavily"),
    }
}

/// Execute a web search using the configured backend.
pub async fn execute(input: &Value, service_config: Option<&ServiceConfig>) -> Result<ToolResult> {
    let query = match input["query"].as_str() {
        Some(q) if !q.trim().is_empty() => q.trim(),
        _ => {
            return Ok(ToolResult {
                tool_use_id: String::new(),
                output: "Missing required parameter: 'query'".to_string(),
                is_error: true,
                tool_name: None,
                file_path: None,
                files_modified: vec![],
                lines_added: 0,
                lines_removed: 0,
            });
        }
    };

    let config = match service_config {
        Some(c) => c,
        None => {
            return Ok(ToolResult {
                tool_use_id: String::new(),
                output: "No search backend configured. Add to your config.toml:\n\n\

/// Converts an `AgentEvent` into the corresponding `CoreEvent`.
impl From<AgentEvent> for CoreEvent {
    fn from(event: AgentEvent) -> Self {
        match event {
            AgentEvent::TextDelta(text) => CoreEvent::TextDelta(text),
            AgentEvent::ThinkingDelta(text) => CoreEvent::ThinkingDelta(text),
            AgentEvent::ToolCall { id, name, arguments } => {
                CoreEvent::ToolCall { id, name, arguments }
            }
            AgentEvent::TurnComplete {
                input_tokens,
                output_tokens,
                cache_read_tokens,
                cache_creation_tokens,
            } => CoreEvent::TurnComplete {
                input_tokens,
                output_tokens,
                cache_read_tokens,
                cache_creation_tokens,
            },
            AgentEvent::Error(msg) => CoreEvent::Error(msg),
            AgentEvent::ProviderError {
                category,
                provider,
                message,
                hint,
            } => CoreEvent::ProviderError {
                category: format!("{:?}", category),
                provider,
                message,
                hint,
            },
            AgentEvent::CompactionComplete => CoreEvent::CompactionComplete,
        }
    }
}

/// Handle for consumers to interact with the core.
#[derive(Clone)]
pub struct CoreHandle {
    /// Send commands to the core.
    pub commands: mpsc::UnboundedSender<CoreCommand>,
    /// Subscribe to events from the core.
    event_tx: broadcast::Sender<CoreEvent>,
}

impl CoreHandle {
    /// Create a new CoreHandle pair (handle + event receiver for the core loop).
    pub fn new() -> (Self, mpsc::UnboundedReceiver<CoreCommand>) {
        let (cmd_tx, cmd_rx) = mpsc::unbounded_channel();
        let (event_tx, _) = broadcast::channel(1024);
        let handle = Self {
            commands: cmd_tx,
            event_tx,
        };
        (handle, cmd_rx)
    }

    /// Subscribe to core events. Returns a broadcast receiver.
    pub fn subscribe(&self) -> broadcast::Receiver<CoreEvent> {
        self.event_tx.subscribe()
    }

    /// Emit an event to all subscribers.
    pub fn emit(&self, event: CoreEvent) {
        // Ignore send errors (no subscribers)
        let _ = self.event_tx.send(event);
    }

    /// Send a command to the core.
    pub fn send(&self, command: CoreCommand) -> Result<(), mpsc::error::SendError<CoreCommand>> {
        self.commands.send(command)
    }
}

//! Background agent manager — spawns and tracks autonomous agent tasks.

use std::collections::HashMap;
use std::sync::Arc;
use tokio::sync::{Mutex, RwLock};
use tokio::task::JoinHandle;

use crate::events::{CoreEvent, CoreHandle};

/// Status of a background agent.
#[derive(Debug, Clone, PartialEq)]
pub enum BackgroundAgentStatus {
    Running,
    Completed,
    Killed,
    BudgetExceeded,
    Error(String),
}

/// Info about a running or completed background agent.
#[derive(Debug, Clone)]
pub struct BackgroundAgentInfo {
    pub id: String,
    pub prompt_summary: String,
    pub status: BackgroundAgentStatus,
    pub tokens_used: u64,
    pub budget: u64,
    pub session_id: String,
    pub parent_session_id: String,
}

/// Configuration for background agent budgets.
#[derive(Debug, Clone)]
pub struct BackgroundAgentConfig {
    pub per_agent_budget: u64,
    pub global_budget: u64,
    pub max_concurrent: usize,
}

impl Default for BackgroundAgentConfig {
    fn default() -> Self {
        Self {
            per_agent_budget: 100_000,
            global_budget: 500_000,
            max_concurrent: 5,
        }
    }
}

/// Manages background agent lifecycle.
pub struct BackgroundAgentManager {
    agents: Arc<RwLock<HashMap<String, BackgroundAgentInfo>>>,
    handles: Arc<Mutex<HashMap<String, JoinHandle<()>>>>,
    config: BackgroundAgentConfig,
    core_handle: CoreHandle,
}

impl BackgroundAgentManager {
    pub fn new(config: BackgroundAgentConfig, core_handle: CoreHandle) -> Self {
        Self {
            agents: Arc::new(RwLock::new(HashMap::new())),
            handles: Arc::new(Mutex::new(HashMap::new())),
            config,
            core_handle,
        }
    }

    /// Get total tokens used across all running agents.
    pub async fn total_tokens_used(&self) -> u64 {
        self.agents
            .read()
            .await
            .values()
            .filter(|a| a.status == BackgroundAgentStatus::Running)
            .map(|a| a.tokens_used)
            .sum()
    }

    /// Get count of running agents.
    pub async fn running_count(&self) -> usize {
        self.agents
            .read()
            .await
            .values()
            .filter(|a| a.status == BackgroundAgentStatus::Running)
            .count()
    }

    /// Check if a new agent can be spawned within budget/concurrency limits.
    pub async fn can_spawn(&self) -> Result<(), String> {
        let running = self.running_count().await;
        if running >= self.config.max_concurrent {
            return Err(format!(
                "Max concurrent agents reached ({}/{}). Kill a running agent first.",
                running, self.config.max_concurrent
            ));
        }

        let total_used = self.total_tokens_used().await;
        if total_used >= self.config.global_budget {
            return Err(format!(
                "Global budget exhausted ({}/{} tokens). Wait for agents to finish or increase budget.",
                total_used, self.config.global_budget
            ));
        }

        Ok(())
    }

    /// Register a new background agent (call before spawning the task).
    pub async fn register(
        &self,
        id: String,
        prompt_summary: String,
        session_id: String,
        parent_session_id: String,
        budget: Option<u64>,
    ) -> BackgroundAgentInfo {
        let budget = budget.unwrap_or(self.config.per_agent_budget);
        let info = BackgroundAgentInfo {
            id: id.clone(),
            prompt_summary: prompt_summary.clone(),
            status: BackgroundAgentStatus::Running,
            tokens_used: 0,
            budget,
            session_id: session_id.clone(),
            parent_session_id: parent_session_id.clone(),
        };
        self.agents.write().await.insert(id.clone(), info.clone());
        self.core_handle.emit(CoreEvent::BackgroundAgentStarted {
            id,
            prompt_summary,
            budget,
            parent_session_id,
        });
        info
    }

    /// Update token usage for a running agent. Returns true if budget exceeded.
    pub async fn update_tokens(&self, id: &str, tokens_used: u64, turn_count: u32) -> bool {
        let mut agents = self.agents.write().await;
        if let Some(agent) = agents.get_mut(id) {
            agent.tokens_used = tokens_used;
            let budget_remaining = agent.budget.saturating_sub(tokens_used);
            self.core_handle
                .emit(CoreEvent::BackgroundAgentProgress {
                    id: id.to_string(),
                    tokens_used,
                    budget_remaining,
                    turn_count,
                });
            return tokens_used >= agent.budget;
        }
        false
    }

    /// Mark an agent as completed.
    pub async fn mark_complete(&self, id: &str) {
        let mut agents = self.agents.write().await;
        if let Some(agent) = agents.get_mut(id) {
            agent.status = BackgroundAgentStatus::Completed;
            self.core_handle
                .emit(CoreEvent::BackgroundAgentComplete {
                    id: id.to_string(),
                    tokens_used: agent.tokens_used,
                    session_id: agent.session_id.clone(),
                });
        }
    }

    /// Mark an agent as failed.
    pub async fn mark_failed(&self, id: &str, reason: &str) {
        let mut agents = self.agents.write().await;
        if let Some(agent) = agents.get_mut(id) {
            agent.status = BackgroundAgentStatus::Error(reason.to_string());
            self.core_handle
                .emit(CoreEvent::BackgroundAgentFailed {
                    id: id.to_string(),
                    reason: reason.to_string(),
                    tokens_used: agent.tokens_used,
                });
        }
    }

    /// Kill a running agent.
    pub async fn kill(&self, id: &str) -> Result<(), String> {
        let mut handles = self.handles.lock().await;
        if let Some(handle) = handles.remove(id) {
            handle.abort();
            let mut agents = self.agents.write().await;
            if let Some(agent) = agents.get_mut(id) {
                agent.status = BackgroundAgentStatus::Killed;
                self.core_handle
                    .emit(CoreEvent::BackgroundAgentFailed {
                        id: id.to_string(),
                        reason: "Killed by user".to_string(),
                        tokens_used: agent.tokens_used,
                    });
            }
            Ok(())
        } else {
            Err(format!("No running agent with id '{id}'"))
        }
    }

    /// Store a task handle for a background agent.
    pub async fn store_handle(&self, id: String, handle: JoinHandle<()>) {
        self.handles.lock().await.insert(id, handle);
    }

    /// List all agents (running and completed).
    pub async fn list(&self) -> Vec<BackgroundAgentInfo> {
        self.agents.read().await.values().cloned().collect()
    }
}

/// Configuration for the server.
pub struct ServerConfig {
    pub port: u16,
    pub bind: String,
    pub config: Config,
}

impl Default for ServerConfig {
    fn default() -> Self {
        Self {
            port: 9090,
            bind: "0.0.0.0".into(),
            config: Config::default(),
        }
    }

/// Handle to the running server.
pub struct ServerHandle {
    pub local_addr: SocketAddr,
    shutdown_tx: Option<oneshot::Sender<()>>,
    pub state: Arc<AppState>,
}

impl ServerHandle {
    /// Shut down the server gracefully.
    pub fn shutdown(mut self) {
        if let Some(tx) = self.shutdown_tx.take() {
            let _ = tx.send(());
        }
    }
}

/// Start the WebSocket server.
pub async fn start_server(
    config: ServerConfig,
    core_handle: CoreHandle,
) -> Result<ServerHandle> {
    let state = AppState::new(core_handle, config.config);
    let app = Router::new().with_state(state.clone());

    let addr: SocketAddr = format!("{}:{}", config.bind, config.port).parse()?;
    let listener = tokio::net::TcpListener::bind(addr).await?;
    let local_addr = listener.local_addr()?;

    let (shutdown_tx, shutdown_rx) = oneshot::channel();

    tokio::spawn(async move {
        tracing::info!("caboose-server listening on {}", local_addr);
        axum::serve(listener, app)
            .with_graceful_shutdown(async {
                let _ = shutdown_rx.await;
            })
            .await
            .ok();
    });

    Ok(ServerHandle {
        local_addr,
        shutdown_tx: Some(shutdown_tx),
        state,
    })
}

//! Device token storage — SQLite-backed paired device management.

use anyhow::{Context, Result};
use argon2::{Argon2, PasswordHash, PasswordHasher, PasswordVerifier};
use argon2::password_hash::SaltString;
use rand::rngs::OsRng;
use rusqlite::Connection;

const MAX_DEVICES: usize = 10;

/// A paired device record.
#[derive(Debug, Clone)]
pub struct Device {
    pub id: String,
    pub name: String,
    pub paired_at: String,
    pub last_seen: Option<String>,
}

/// Manages device tokens in SQLite.
pub struct DeviceStore {
    db_path: std::path::PathBuf,
}

impl DeviceStore {
    pub fn new(db_path: std::path::PathBuf) -> Result<Self> {
        let store = Self { db_path };
        store.init_table()?;
        Ok(store)
    }

    fn connect(&self) -> Result<Connection> {
        Connection::open(&self.db_path).context("Failed to open device database")
    }

    fn init_table(&self) -> Result<()> {
        let conn = self.connect()?;
        conn.execute_batch(
            "CREATE TABLE IF NOT EXISTS devices (
                id            TEXT PRIMARY KEY,
                name          TEXT NOT NULL,
                token_prefix  TEXT NOT NULL,
                token_hash    TEXT NOT NULL,
                paired_at     TEXT NOT NULL,
                last_seen     TEXT,
                revoked       INTEGER DEFAULT 0
            );
            CREATE INDEX IF NOT EXISTS idx_devices_token_prefix
                ON devices(token_prefix) WHERE revoked = 0;",
        )?;
        Ok(())
    }

    /// Count active (non-revoked) devices.
    pub fn active_count(&self) -> Result<usize> {
        let conn = self.connect()?;
        let count: usize = conn.query_row(
            "SELECT COUNT(*) FROM devices WHERE revoked = 0",
            [],
            |row| row.get(0),
        )?;
        Ok(count)
    }

    /// Store a new paired device. Returns the raw token.
    pub fn pair(&self, device_id: &str, device_name: &str) -> Result<String> {
        if self.active_count()? >= MAX_DEVICES {
            anyhow::bail!("Maximum devices reached ({MAX_DEVICES}). Unpair a device first.");
        }

        let token = generate_token();
        let prefix = &token[..16]; // first 8 bytes = 16 hex chars
        let hash = hash_token(&token)?;
        let now = chrono::Utc::now().to_rfc3339();

        let conn = self.connect()?;
        conn.execute(
            "INSERT INTO devices (id, name, token_prefix, token_hash, paired_at) VALUES (?1, ?2, ?3, ?4, ?5)",
            rusqlite::params![device_id, device_name, prefix, hash, now],
        )?;

        Ok(token)
    }

    /// Verify a device token. Returns the device if valid.
    pub fn verify(&self, token: &str) -> Result<Option<Device>> {
        if token.len() < 16 {
            return Ok(None);
        }
        let prefix = &token[..16];
        let conn = self.connect()?;

        let mut stmt = conn.prepare(
            "SELECT id, name, token_hash, paired_at, last_seen FROM devices WHERE token_prefix = ?1 AND revoked = 0",
        )?;

        let candidates: Vec<(String, String, String, String, Option<String>)> = stmt
            .query_map([prefix], |row| {
                Ok((
                    row.get(0)?,
                    row.get(1)?,
                    row.get(2)?,
                    row.get(3)?,
                    row.get(4)?,
                ))
            })?
            .filter_map(|r| r.ok())
            .collect();

        for (id, name, hash, paired_at, last_seen) in candidates {
            if verify_token(token, &hash) {
                // Update last_seen
                let now = chrono::Utc::now().to_rfc3339();
                conn.execute(
                    "UPDATE devices SET last_seen = ?1 WHERE id = ?2",
                    rusqlite::params![now, id],
                )?;
                return Ok(Some(Device {
                    id,
                    name,
                    paired_at,
                    last_seen: Some(now),
                }));
            }
        }

        Ok(None)
    }

    /// Revoke a device by ID.
    pub fn revoke(&self, device_id: &str) -> Result<bool> {
        let conn = self.connect()?;
        let rows = conn.execute(
            "UPDATE devices SET revoked = 1 WHERE id = ?1 AND revoked = 0",
            [device_id],
        )?;
        Ok(rows > 0)
    }

    /// List all active devices.
    pub fn list(&self) -> Result<Vec<Device>> {
        let conn = self.connect()?;
        let mut stmt = conn.prepare(
            "SELECT id, name, paired_at, last_seen FROM devices WHERE revoked = 0 ORDER BY paired_at DESC",
        )?;
        let devices = stmt
            .query_map([], |row| {
                Ok(Device {
                    id: row.get(0)?,
                    name: row.get(1)?,
                    paired_at: row.get(2)?,
                    last_seen: row.get(3)?,
                })
            })?
            .filter_map(|r| r.ok())
            .collect();
        Ok(devices)
    }
}

/// Generate a 256-bit random token as hex string.
fn generate_token() -> String {
    use rand::Rng;
    let bytes: [u8; 32] = rand::thread_rng().gen();
    hex::encode(bytes)
}

/// Hash a token with argon2.
fn hash_token(token: &str) -> Result<String> {
    let salt = SaltString::generate(&mut OsRng);
    let argon2 = Argon2::default();
    let hash = argon2
        .hash_password(token.as_bytes(), &salt)
        .map_err(|e| anyhow::anyhow!("Hash failed: {e}"))?;
    Ok(hash.to_string())
}

/// Verify a token against an argon2 hash.
fn verify_token(token: &str, hash: &str) -> bool {
    let parsed = match PasswordHash::new(hash) {
        Ok(h) => h,
        Err(_) => return false,
    };
    Argon2::default()
        .verify_password(token.as_bytes(), &parsed)
        .is_ok()
}

//! One-time pairing code generation and validation.

use rand::Rng;
use std::time::{Duration, Instant};

const CODE_LENGTH: usize = 6;
const CODE_EXPIRY: Duration = Duration::from_secs(300); // 5 minutes

/// A one-time pairing code.
#[derive(Debug)]
pub struct PairingCode {
    pub code: String,
    created_at: Instant,
    used: bool,
}

impl PairingCode {
    pub fn is_expired(&self) -> bool {
        self.created_at.elapsed() > CODE_EXPIRY
    }

    pub fn is_valid(&self) -> bool {
        !self.used && !self.is_expired()
    }
}

/// Manages active pairing codes.
pub struct PairingManager {
    codes: Vec<PairingCode>,
}

impl PairingManager {
    pub fn new() -> Self {
        Self { codes: Vec::new() }
    }

    /// Generate a new 6-character alphanumeric pairing code.
    pub fn generate(&mut self) -> String {
        // Clean up expired codes
        self.codes.retain(|c| !c.is_expired());

        let code = random_code();
        self.codes.push(PairingCode {
            code: code.clone(),
            created_at: Instant::now(),
            used: false,
        });
        code
    }

    /// Validate and burn a pairing code. Returns true if valid.
    pub fn validate(&mut self, code: &str) -> bool {
        if let Some(entry) = self.codes.iter_mut().find(|c| c.code == code && c.is_valid()) {
            entry.used = true;
            true
        } else {
            false
        }
    }
}

fn random_code() -> String {
    const CHARS: &[u8] = b"ABCDEFGHJKLMNPQRSTUVWXYZ23456789"; // no I/O/0/1 for clarity
    let mut rng = rand::thread_rng();
    (0..CODE_LENGTH)
        .map(|_| CHARS[rng.gen_range(0..CHARS.len())] as char)
        .collect()
}

//! Maps CoreEvent to JSON and incoming JSON to CoreCommand.

use caboose_core::events::{CoreCommand, CoreEvent};
use serde_json::{json, Value};

use crate::ws::envelope::{IncomingMessage, OutgoingMessage};

/// Convert a CoreEvent into an OutgoingMessage.
pub fn event_to_message(event: &CoreEvent, id: &str) -> OutgoingMessage {
    match event {
        CoreEvent::TextDelta(text) => {
            OutgoingMessage::event(id, "TextDelta", json!({"text": text}))
        }
        CoreEvent::ThinkingDelta(text) => {
            OutgoingMessage::event(id, "ThinkingDelta", json!({"text": text}))
        }
        CoreEvent::ToolCall { id: tid, name, arguments } => {
            OutgoingMessage::event(id, "ToolCall", json!({
                "id": tid, "name": name, "arguments": arguments
            }))
        }
        CoreEvent::TurnComplete { input_tokens, output_tokens, cache_read_tokens, cache_creation_tokens } => {
            OutgoingMessage::event(id, "TurnComplete", json!({
                "input_tokens": input_tokens,
                "output_tokens": output_tokens,
                "cache_read_tokens": cache_read_tokens,
                "cache_creation_tokens": cache_creation_tokens,
            }))
        }
        CoreEvent::Error(msg) => {
            OutgoingMessage::event(id, "Error", json!({"message": msg}))
        }
        CoreEvent::ToolApprovalRequired { tool_calls, current_index } => {
            let calls: Vec<Value> = tool_calls.iter().map(|tc| json!({
                "id": tc.id, "name": tc.name, "arguments": tc.arguments,
            })).collect();
            OutgoingMessage::event(id, "ToolApprovalRequired", json!({
                "tool_calls": calls, "index": current_index
            }))
        }
        CoreEvent::BackgroundAgentStarted { id: aid, prompt_summary, budget, parent_session_id } => {
            OutgoingMessage::event(id, "BackgroundAgentStarted", json!({
                "id": aid, "prompt_summary": prompt_summary,
                "budget": budget, "parent_session_id": parent_session_id,
            }))
        }
        CoreEvent::BackgroundAgentProgress { id: aid, tokens_used, budget_remaining, turn_count } => {
            OutgoingMessage::event(id, "BackgroundAgentProgress", json!({
                "id": aid, "tokens_used": tokens_used,
                "budget_remaining": budget_remaining, "turn_count": turn_count,
            }))
        }
        CoreEvent::BackgroundAgentComplete { id: aid, tokens_used, session_id } => {
            OutgoingMessage::event(id, "BackgroundAgentComplete", json!({
                "id": aid, "tokens_used": tokens_used, "session_id": session_id,
            }))
        }
        CoreEvent::BackgroundAgentFailed { id: aid, reason, tokens_used } => {
            OutgoingMessage::event(id, "BackgroundAgentFailed", json!({
                "id": aid, "reason": reason, "tokens_used": tokens_used,
            }))
        }
        // Remaining event variants — mapped with minimal payloads for now.
        // Full payload mappings for session, MCP, provider, checkpoint, roundhouse
        // events will be added as the mobile app consumes them.
        CoreEvent::CompactionComplete => OutgoingMessage::event(id, "CompactionComplete", json!({})),
        CoreEvent::ToolExecuted(result) => OutgoingMessage::event(id, "ToolExecuted", json!({
            "tool_use_id": result.tool_use_id,
            "output": result.output,
            "is_error": result.is_error,
        })),
        CoreEvent::SessionCreated(session) => OutgoingMessage::event(id, "SessionCreated", json!({"id": session.id})),
        CoreEvent::SessionList(sessions) => OutgoingMessage::event(id, "SessionList", json!({
            "sessions": sessions.iter().map(|s| json!({"id": s.id, "title": s.title})).collect::<Vec<_>>()
        })),
        CoreEvent::SessionDeleted { session_id } => OutgoingMessage::event(id, "SessionDeleted", json!({"session_id": session_id})),
        CoreEvent::ProviderSwitched { provider, model } => OutgoingMessage::event(id, "ProviderSwitched", json!({"provider": provider, "model": model})),
        CoreEvent::Status { provider, model, session_id, permission_mode } => OutgoingMessage::event(id, "Status", json!({
            "provider": provider, "model": model, "session_id": session_id, "permission_mode": permission_mode,
        })),
        _ => OutgoingMessage::event(id, "Unhandled", json!({})),
    }
}

/// Parse an incoming command message into a CoreCommand.
pub fn message_to_command(msg: &IncomingMessage) -> Result<CoreCommand, String> {
    let command = msg.command.as_deref().ok_or("Missing 'command' field")?;
    let payload = msg.payload.as_ref().cloned().unwrap_or(Value::Null);

    match command {
        "SendMessage" => {
            let text = payload["text"]
                .as_str()
                .ok_or("SendMessage requires 'text' field")?
                .to_string();
            Ok(CoreCommand::SendMessage { text })
        }
        "CancelTurn" => Ok(CoreCommand::CancelTurn),
        "ApproveTool" => Ok(CoreCommand::ApproveTool),
        "DenyTool" => Ok(CoreCommand::DenyTool),
        "AlwaysAllowTool" => Ok(CoreCommand::AlwaysAllowTool),
        "CreateSession" => Ok(CoreCommand::CreateSession),
        "ListSessions" => {
            let limit = payload["limit"].as_u64().unwrap_or(20) as usize;
            Ok(CoreCommand::ListSessions { limit })
        }
        "LoadSession" => {
            let session_id = payload["session_id"]
                .as_str()
                .ok_or("LoadSession requires 'session_id'")?
                .to_string();
            Ok(CoreCommand::LoadSession { session_id })
        }
        "DeleteSession" => {
            let session_id = payload["session_id"]
                .as_str()
                .ok_or("DeleteSession requires 'session_id'")?
                .to_string();
            Ok(CoreCommand::DeleteSession { session_id })
        }
        "GetStatus" => Ok(CoreCommand::GetStatus),
        "SpawnBackgroundAgent" => {
            let prompt = payload["prompt"]
                .as_str()
                .ok_or("SpawnBackgroundAgent requires 'prompt'")?
                .to_string();
            let budget = payload["budget"].as_u64();
            Ok(CoreCommand::SpawnBackgroundAgent { prompt, budget })
        }
        "KillBackgroundAgent" => {
            let id = payload["id"]
                .as_str()
                .ok_or("KillBackgroundAgent requires 'id'")?
                .to_string();
            Ok(CoreCommand::KillBackgroundAgent { id })
        }
        other => Err(format!("Unknown command: '{other}'")),
    }
}

//! Per-connection WebSocket session actor.

use std::sync::Arc;
use axum::extract::ws::{Message, WebSocket};
use futures_util::{SinkExt, StreamExt};
use tokio::sync::broadcast;

use caboose_core::events::CoreEvent;
use crate::bridge;
use crate::state::AppState;
use crate::ws::envelope::IncomingMessage;

/// Handle a single authenticated WebSocket connection.
pub async fn handle_session(
    mut socket: WebSocket,
    state: Arc<AppState>,
    device_id: String,
) {
    let mut event_rx = state.core_handle.subscribe();

    loop {
        tokio::select! {
            // Forward core events to client
            Ok(event) = event_rx.recv() => {
                let msg = bridge::event_to_message(&event, "");
                let json = serde_json::to_string(&msg).unwrap_or_default();
                if socket.send(Message::Text(json.into())).await.is_err() {
                    break;
                }
            }
            // Receive commands from client
            Some(Ok(msg)) = socket.next() => {
                match msg {
                    Message::Text(text) => {
                        if let Ok(incoming) = serde_json::from_str::<IncomingMessage>(&text) {
                            match bridge::message_to_command(&incoming) {
                                Ok(cmd) => {
                                    let _ = state.core_handle.send(cmd);
                                }
                                Err(e) => {
                                    let err = crate::ws::envelope::OutgoingMessage::error(&incoming.id, &e);
                                    let json = serde_json::to_string(&err).unwrap_or_default();
                                    let _ = socket.send(Message::Text(json.into())).await;
                                }
                            }
                        }
                    }
                    Message::Close(_) => break,
                    _ => {}
                }
            }
            else => break,
        }
    }
}
